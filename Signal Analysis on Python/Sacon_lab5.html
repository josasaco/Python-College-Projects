def ideal_lpf(wc, M):
    alpha = (M-1)/2
    n = np.arange(0,M)
    m = n - alpha
    fc = wc / np.pi
    h = fc * np.sinc(fc*m)
    return h

def freqz_mod(b,a):
    [w, H] = signal.freqz(b,a,1000,'True') 
    H = np.transpose(H[0:501])
    w = np.transpose(w[0:501])
    magH = np.abs(H)
    dB = 20 * np.log10(magH/magH.max())
    phase = np.angle(H)
    grpdelay = signal.group_delay((b,a))
    return dB, phase, grpdelay, w
